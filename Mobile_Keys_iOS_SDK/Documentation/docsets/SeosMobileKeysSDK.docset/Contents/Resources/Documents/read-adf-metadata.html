Read ADF metadata

The first method reads the meta tag using no encryption keys

- (void)testGetDataFromMetadataTag {
    NSLog(@"Read tag FF41 from all keys using no encryption");
    NSError *keysError = nil;
    NSArray<MobileKeysKey *> *keys = [_mobileKeysManager listMobileKeys:&keysError];

    MobileKeysSeosProvider *seosProvider = [_mobileKeysManager defaultMobileKeysSeosProvider];

    // Read metadata from all keys
    for (MobileKeysKey *key in keys) {

        // Session parameters
        MobileKeysSessionParameters *sessionParameters = [[MobileKeysSessionParameters alloc] initWithMobileKeysSelectionType:MobileKeysSelectionExtendedADF];
        [sessionParameters setRequireSelectAid:YES];
        [sessionParameters setContactless:NO];
        [sessionParameters setUseTestVectors:NO];
        NSData *oid = [self hexToData:key.keyId];
        [sessionParameters setOidSelectionList:@[oid]];

        // Open Seos Session
        NSError *error = nil;
        NSData *selectedOid = [seosProvider openSessionWithParams:sessionParameters withError:&error];

        // What was selected?
        NSLog(@"Selected OID: %@", selectedOid);

        // Read tag FF41 - Metadata
        MobileKeysSeosTag *tag = [[MobileKeysSeosTag alloc] initWithTagNumber:0xFF41];
        MobileKeysGetDataApduCommand *getData = [[MobileKeysGetDataApduCommand alloc] initWithTag:tag];
        NSError *processError = nil;
        MobileKeysApduResponse *response = [seosProvider processApduCommand:getData withError:&processError];
        NSLog(@"GetData returned StatusWord %@", [self dataToHex:response.statusWordData]);
        NSLog(@"GetData returned Data %@", [self dataToHex:response.data]);

        // Close Seos Session
        [seosProvider closeSeosSession:&error];
    }
}
