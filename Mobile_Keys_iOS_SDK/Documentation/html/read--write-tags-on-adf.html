This below test code can be implemented in the SeosExampleApplication. Run this code in SeosExampleApplication MobileKeysController after a specific test credential is issued. This specific test credential with plain keys is is include here too.

This method reads and writes data to tag 0xFF01 using encryption keys

- (void)testSeosAccessApi {

    NSError *keysError = nil;
    NSArray<MobileKeysKey *> *keys = [_mobileKeysManager listMobileKeys:&keysError];

    MobileKeysSeosProvider seosProvider = [_mobileKeysManager defaultMobileKeysSeosProvider];

    // Read/writing tag 0xFF01 from all keysusing encryption
    for (MobileKeysKey *key in keys) {

        // Select the key
        NSData *oid = [self hexToData:key.keyId];
        NSArray *oids = @[oid];

        // Session parameters
        MobileKeysSymmetricKeyPair *privacyKey = [[MobileKeysSymmetricKeyPair alloc] initWithEncKey:[self hexToData:@"5A5A5A5A5A5A5A5A5A5A5A5A5A5A5A5A"]
                                                                                             macKey:[self hexToData:@"8AA83BF8CBDA10620BC1BF19FBB6CD58"]];

        MobileKeysPrivacyKeySet *privacyKeySet = [[MobileKeysPrivacyKeySet alloc] initWithKeyNumber:MOBILE_KEYS_KEY_0
                                                                                   symmetricKeyPair:privacyKey
                                                                                encryptionAlgorithm:MOBILE_KEYS_ENCRYPTION_ALGO_AES_128];

        // Auth key one used for reading
        MobileKeysSymmetricKeyPair *authKeyOne = [[MobileKeysSymmetricKeyPair alloc] initWithEncKey:[self hexToData:@"E0F75D03506A735C93DF39113D18532F"]
                                                                                             macKey:[self hexToData:@"F5582EADD895225DC71D080E19338A25"]];

        MobileKeysAuthenticationKeySet *authenticationKeyOne = [[MobileKeysAuthenticationKeySet alloc] initWithKeyNumber:MOBILE_KEYS_KEY_1
                                                                                                        symmetricKeyPair:authKeyOne
                                                                                                     encryptionAlgorithm:MOBILE_KEYS_ENCRYPTION_ALGO_AES_128 global:NO];

        // Auth key two used for writing
        MobileKeysSymmetricKeyPair *authKeyTwo = [[MobileKeysSymmetricKeyPair alloc] initWithEncKey:[self hexToData:@"269C555F45BF3EE0095EA1E385713131"]
                                                                                             macKey:[self hexToData:@"11BDB88E534FF02CA4E77E858DCCD37E"]];

        MobileKeysAuthenticationKeySet *authenticationKeyTwo = [[MobileKeysAuthenticationKeySet alloc] initWithKeyNumber:MOBILE_KEYS_KEY_2
                                                                                                        symmetricKeyPair:authKeyTwo
                                                                                                     encryptionAlgorithm:MOBILE_KEYS_ENCRYPTION_ALGO_AES_128 global:NO];

        // Setup session parameters for use with auth key one
        MobileKeysSessionParameters *sessionParamOne = [[MobileKeysSessionParameters alloc] initWithMobileKeysSelectionType:MobileKeysSelectionADF
                                                                                                    mobileKeysPrivacyKeySet:privacyKeySet
                                                                                             mobileKeysAuthenticationKeySet:authenticationKeyOne];
        [sessionParamOne setRequireSelectAid:YES];
        [sessionParamOne setContactless:NO];
        [sessionParamOne setUseTestVectors:NO];
        [sessionParamOne setOidSelectionList:oids];

        // Setup session parameters for use with auth key two
        MobileKeysSessionParameters *sessionParamTwo = [[MobileKeysSessionParameters alloc] initWithMobileKeysSelectionType:MobileKeysSelectionADF
                                                                                                    mobileKeysPrivacyKeySet:privacyKeySet
                                                                                             mobileKeysAuthenticationKeySet:authenticationKeyTwo];
        [sessionParamTwo setRequireSelectAid:YES];
        [sessionParamTwo setContactless:YES];
        [sessionParamTwo setUseTestVectors:NO];
        [sessionParamTwo setOidSelectionList:oids];

        NSLog(@"Test of reading and writing to tag 0xFF01");

        NSError *processError = nil;
        NSError *error = nil;

        // Open Seos Session with auth key 2 to write to tag 0xFF01
        NSData *selectedOid = [seosProvider openSessionWithParams:sessionParamTwo withError:&error];

        // What was selected?
        NSLog(@"Selected OID: %@", selectedOid);

        // Tag 0xFF01
        MobileKeysSeosTag *tagFF01 = [[MobileKeysSeosTag alloc] initWithTagNumber:0xFF01];

        // Write to tag 0xFF01 with auth key two
        MobileKeysPutDataApduCommand *putData = [[MobileKeysPutDataApduCommand alloc] initWithTag:tagFF01 andData:[self hexToData:@"0402ABCD"] secureMessaging:YES];
        MobileKeysApduResponse *responseFromPut = [seosProvider processApduCommand:putData withError:&processError];
        NSLog(@"Writing 0402ABCD to tag 0xFF01");
        NSLog(@"PutData returned StatusWord %@", responseFromPut.statusWordData);

        // Close Seos Session
        [seosProvider closeSeosSession:&error];
        // Start Seos session with new parms
        [seosProvider openSessionWithParams:sessionParamOne withError:&error];

        // Read tag 0xFF01 with auth key one
        MobileKeysGetDataApduCommand *getData = [[MobileKeysGetDataApduCommand alloc] initWithTag:tagFF01 secureMessaging:YES];
        MobileKeysApduResponse *responseRead = [seosProvider processApduCommand:getData withError:&processError];
        NSLog(@"Read tag 0xFF01. Data = %@", responseRead.data);
        NSLog(@"GetData returned StatusWord %@", responseRead.statusWordData);

        // Close Seos Session
        [seosProvider closeSeosSession:&error];
        // Start Seos session with new parms
        [seosProvider openSessionWithParams:sessionParamTwo withError:&error];

        // Write to tag 0xFF01 with auth key two
        MobileKeysPutDataApduCommand *putDataAgain = [[MobileKeysPutDataApduCommand alloc] initWithTag:tagFF01 andData:[self hexToData:@"0402BEEF"] secureMessaging:YES];
        MobileKeysApduResponse *responseFromPutAgain = [seosProvider processApduCommand:putDataAgain withError:&processError];
        NSLog(@"Writing 0402BEEF to tag 0xFF01");
        NSLog(@"PutData returned StatusWord %@", responseFromPutAgain.statusWordData);

        // Close Seos Session
        [seosProvider closeSeosSession:&error];
        // Start Seos session with new parms
        [seosProvider openSessionWithParams:sessionParamOne withError:&error];

        // Read tag 0xFF01 with auth key one again to see change of data
        MobileKeysGetDataApduCommand *getDataAgain = [[MobileKeysGetDataApduCommand alloc] initWithTag:tagFF01 secureMessaging:YES];
        MobileKeysApduResponse *responseReadAgain = [seosProvider processApduCommand:getDataAgain withError:&processError];
        NSLog(@"Read tag 0xFF01. Data = %@", responseReadAgain.data);
        NSLog(@"GetData returned Statusword %@", responseReadAgain.statusWordData);

        // Close Seos Session
        [seosProvider closeSeosSession:&error];
    }
}

Helper methods for converting HEX to NSData

- (NSString *)dataToHex:(NSData *)data {
    const unsigned char *dataBuffer = (const unsigned char *) [data bytes];

    if (!dataBuffer)
        return [NSString string];

    NSUInteger dataLength = [data length];
    NSMutableString *hexString = [NSMutableString stringWithCapacity:(dataLength * 2)];

    for (NSUInteger i = 0; i < dataLength; ++i)
        [hexString appendString:[NSString stringWithFormat:@"%02X", (unsigned int) dataBuffer[i]]];

    return hexString;
}

- (NSData *)hexToData:(NSString *)hex {
    NSMutableData *data = [[NSMutableData alloc] init];
    NSUInteger idx;
    for (idx = 0; idx + 2 <= hex.length; idx += 2) {
        NSRange range = NSMakeRange(idx, 2);
        NSString *hexStr = [hex substringWithRange:range];
        NSScanner *scanner = [NSScanner scannerWithString:hexStr];
        unsigned int intValue;
        [scanner scanHexInt:&intValue];
        char bytesToAppend[1] = {intValue};
        [data appendBytes:bytesToAppend length:1];
    }
    return data;
}
