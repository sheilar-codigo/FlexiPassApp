Master authentication keyset

You can use this sample code to test the master authentication keyset functionality. The sample code will personalize an endpoint using an AMR with a ADF with a master key.

The sample code will then setup a new 

- (void)testMasterAuthenticationKeySetAuthentication {
    NSData *gdfPrivacyEncryptionKey = [self hexToData:@"73D51ABBD89CB8196F0EFB6892F94D68"];
    NSData *gdfPrivacyMacKey = [self hexToData:@"FCCC2C35F0B84609E5F12C55DD85ABA8"];
    NSData *gdfDiversifier = [self hexToData:@"BA5BB5F67E1BDA28"];
    NSData *gdfAuthKeyOne = [self hexToData:@"D5D9BEF76808F3B572E5900112B81927"];
    NSData *oid = [self hexToData:@""];

    MobileKeysSymmetricKeyPair *_keyPair [[MobileKeysSymmetricKeyPair alloc] initWithEncKey:gdfPrivacyEncryptionKey macKey:gdfPrivacyMacKey];
    MobileKeysSeosProvider *_mobileKeysSeosProvider = [_mobileKeysManager defaultMobileKeysSeosProvider];

    NSData *amrCommandData = [self hexToData:@"00410C03BF618200BB0600CF08BA5BB5F67E1BDA28FF6E14DF010111DF0201A3DF400191DF440191DF450191FF6F1B06010241020105DF010111DF0201A3DF400191DF440191DF450191DF30318773D51ABBD89CB8196F0EFB6892F94D68FCCC2C35F0B84609E5F12C55DD85ABA800000000000000000000000000000000DF313286009965DCAEF215D31C15F1386206157F2A195CCDF5EAE9F62873C55C13E2087FB700000000000000000000000000000000DF01050000000000DF0206000000000000"];
    MobileKeysApduCommand *amrCommand = [[MobileKeysApduCommand alloc] initWithRawData:amrCommandData];

    _mobileKeysSeosProvider = [_mobileKeysManager defaultMobileKeysSeosProvider];
    MobileKeysSessionParameters *params = [[MobileKeysSessionParameters alloc] initWithMobileKeysSelectionType:MobileKeysSelectionEmptySeos mobileKeysPrivacyKeySet:[[MobileKeysPrivacyKeySet alloc] initWithWithoutPrivacyWithKeyNumber:MOBILE_KEYS_KEY_0]];
    params.requireSelectAid = YES;

    NSError *error = nil;
    [_mobileKeysSeosProvider openSessionWithParams:params withError:&error];

    MobileKeysApduResponse *responseFromAmr = [_mobileKeysSeosProvider processApduCommand:amrCommand withError:&error];

    MobileKeysMasterAuthenticationKeySet *masterAuthenticationKeySet = [[MobileKeysMasterAuthenticationKeySet alloc] initWithKeyNumber:MOBILE_KEYS_KEY_17 masterKeyData:gdfAuthKeyOne oid:oid diversifier:gdfDiversifier encryptionAlgorithm:MOBILE_KEYS_ENCRYPTION_ALGO_AES_128 global:NO];

    MobileKeysPrivacyKeySet *plainPrivacyKeySet = [[MobileKeysPrivacyKeySet alloc] initWithKeyNumber:MOBILE_KEYS_KEY_16 symmetricKeyPair:_keyPair encryptionAlgorithm:MOBILE_KEYS_ENCRYPTION_ALGO_AES_128];
    params = [[MobileKeysSessionParameters alloc] initWithMobileKeysSelectionType:MobileKeysSelectionGDF mobileKeysPrivacyKeySet:plainPrivacyKeySet mobileKeysAuthenticationKeySet: masterAuthenticationKeySet];
    params.requireSelectAid = YES;

    MobileKeysGetDataApduCommand *getDataApduCommand = [[MobileKeysGetDataApduCommand alloc] initWithTag:[[MobileKeysSeosTag alloc] initWithTagNumber:0xDF01] secureMessaging:YES];
    MobileKeysApduResponse *responseFromGetData = [_mobileKeysSeosProvider processApduCommand:getDataApduCommand withError:&error];

    //TAG: DF01 LENGTH: 05 VALUE: 00 00 00 00 00
    NSAssert([[self hexToData:@"DF01050000000000"] isEqualToData:responseFromGetData.data], @"The value on DF01 should be DF01050000000000");
}
